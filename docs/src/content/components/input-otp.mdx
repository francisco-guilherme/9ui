## Usage

```tsx title="Imports"
import {
  InputOTP,
  InputOTPGroup,
  InputOTPSeparator,
  InputOTPSlot,
} from "@9ui/uiinput-otp";
```

```tsx title="Anatomy"
<InputOTP>
  <InputOTPGroup>
    <InputOTPSlot />
  </InputOTPGroup>
  <InputOTPSeparator />
</InputOTP>
```
