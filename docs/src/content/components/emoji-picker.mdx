## Usage

```tsx title="Imports"
import {
  EmojiPicker,
  EmojiPickerContent,
  EmojiPickerEmpty,
  EmojiPickerList,
  EmojiPickerLoading,
  EmojiPickerSearch,
} from "@9ui/uiemoji-picker";
```

```tsx title="Anatomy"
<EmojiPicker>
  <EmojiPickerSearch />
  <EmojiPickerContent>
    <EmojiPickerLoading />
    <EmojiPickerEmpty />
    <EmojiPickerList />
  </EmojiPickerContent>
</EmojiPicker>
```
