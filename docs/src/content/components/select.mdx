## Usage

```tsx title="Imports"
import {
  Select,
  SelectContent,
  SelectGroup,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@9ui/uiselect";
```

```tsx title="Anatomy"
<Select>
  <SelectTrigger>
    <SelectValue />
  </SelectTrigger>
  <SelectContent>
    <SelectGroup>
      <SelectGroupLabel />
      <SelectItem />
    </SelectGroup>
  </SelectContent>
</Select>
```
