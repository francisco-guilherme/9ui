## Usage

```tsx title="Imports"
import {
  Carousel,
  CarouselContent,
  CarouselItem,
  CarouselNavigation,
  CarouselNext,
  CarouselPrevious,
} from "@9ui/uicarousel";
```

```tsx title="Anatomy"
<Carousel>
  <CarouselContent>
    <CarouselItem />
  </CarouselContent>
  <CarouselNavigation>
    <CarouselPrevious />
    <CarouselNext />
  </CarouselNavigation>
</Carousel>
```
