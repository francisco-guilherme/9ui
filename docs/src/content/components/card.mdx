## Usage

```tsx title="Imports"
import {
  Card,
  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
} from "@9ui/uicard";
```

```tsx title="Anatomy"
<Card>
	<CardHeader>
		<CardTitle />
		<CardDescription />
	</CardHeader>
	<CardContent>
	<CardFooter />
</Card>
```
