## Usage

```tsx title="Imports"
import {
  Sheet,
  SheetClose,
  SheetContent,
  SheetDescription,
  SheetFooter,
  SheetHeader,
  SheetTitle,
  SheetTrigger,
} from "@9ui/uisheet";
```

```tsx title="Anatomy"
<Sheet>
  <SheetTrigger />
  <SheetContent>
    <SheetClose />
    <SheetHeader>
      <SheetTitle />
      <SheetDescription />
    </SheetHeader>
    <SheetFooter>
      <SheetClose />
    </SheetFooter>
  </SheetContent>
</Sheet>
```
