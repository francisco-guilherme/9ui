## Usage

```tsx title="Imports"
import {
  CodeBar,
  CodeBarButton,
  CodeBarGroup,
  CodeBarInput,
  CodeBarLink,
  CodeBarSeparator,
} from "@9ui/uiCodeBar";
```

```tsx title="Anatomy"
<CodeBar>
  <CodeBarButton />
  <CodeBarLink />
  <CodeBarSeparator />
  <CodeBarGroup>
    <CodeBarButton />
    <CodeBarButton />
  <CodeBarGroup />
  <CodeBarInput />
</CodeBar>
```
