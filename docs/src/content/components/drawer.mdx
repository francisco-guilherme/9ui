## Usage

```tsx title="Imports"
import {
  Drawer,
  DrawerClose,
  DrawerContent,
  DrawerDescription,
  DrawerFooter,
  DrawerHeader,
  DrawerTitle,
  DrawerTrigger,
} from "@9ui/uidrawer";
```

```tsx title="Anatomy"
<Drawer>
	<DrawerTrigger>
	<DrawerContent>
		<DrawerHeader>
			<DrawerTitle />
			<DrawerDescription />
		</DrawerHeader>
		<DrawerFooter>
			<DrawerClose />
		</DrawerFooter>
	</DrawerContent>
</Drawer>
```
