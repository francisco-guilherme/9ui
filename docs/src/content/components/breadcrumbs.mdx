## Usage

```tsx title="Imports"
import {
  Breadcrumb,
  BreadcrumbItem,
  BreadcrumbList,
  BreadcrumbSeparator,
} from "@9ui/uibreadcrumbs";
```

```tsx title="Anatomy"
<Breadcrumb>
  <BreadcrumbList>
    <BreadcrumbItem />
    <BreadcrumbSeparator />
  </BreadcrumbList>
</Breadcrumb>
```
